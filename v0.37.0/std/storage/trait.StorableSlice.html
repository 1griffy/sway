Trait std::storage::StorableSlice

pub trait StorableSlice<T> {
    #[storage(write)]
    fn store(self, argument: T);
    #[storage(read)]
    fn load(self) -> Option<T>;
    #[storage(read, write)]
    fn clear(self) -> bool;
    #[storage(read)]
    fn len(self) -> u64;
}
